# 完成以下算法题

给出思路后实现
var getIntersectionNode = function(headA, headB) {
  let p = headA
  
  while (p) {
    let pb = headB
    while (pb) {
       if (p === pb) {
        return p
       }
       pb = pb.next
    }
    p = p.next
  }
  return null
};
